 ii)(– 5, 7),(– 1, 3).\\
\solution:\\
Given,
\begin{align}
\vec{A}&=\myvec{-5\\7}\\
\vec{B}&=\myvec{-1\\3}\\
\vec{AB}&=\myvec{4\\-4}\\
|AB|&=\sqrt{(4)^2+(-4)^2}\\
|AB|&=\sqrt{16+16}\\
|AB|&=\sqrt{32}\\
|AB|&=2\sqrt{4}\\
\end{align}
\end{enumerate}
